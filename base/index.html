mcGrid

mcGrid

基于angular的一个表格组件，带增删改、分页、搜索组合件

Getting Started

快速指引 (可以结合右边的例子):

  1. 载入 jQuery 和 AngularJS.
  2. 依赖 UI Bootstrap 和 bootstrap-datetimepicker（后期可以考虑加入select2）.
  3. 你声明你的应用程序模块,添加ui.mc.grid、ui.bootstrap:
    angular.module('myApp',['ui.mc.grid','ui.bootstrap']);
    如果你使用 angular seed, 那就加入到 app.js 文件中.
  4. 在 controller 内的 html 里面嵌入:
    <!-- 表格内容 -->
    <div mc-grid options="gridOptions" source="dataSource"></div>
    options：用于配置表格的字段、按钮组、分页.
    source：渲染表格的数据（包含用于分页显示的total）.
  5. 在你 javascript 文件中，嵌入：
    6. $scope.dataSource = [{name: "Moroni", age: 50},
                     {name: "Tiancum", age: 43},
                     {name: "Jacob", age: 27},
                     {name: "Nephi", age: 29},
                     {name: "Enos", age: 34}];
					 
$scope.gridOptions = {columnDefs: [
                          {key: "name", name: "姓名", type: "text"},
                          {key: "age", name: "年龄", type: "text"}
                     ]};
  6. 这就是表格的简单使用，如右边的例子:

基础例子

  • JavaScript 
<html ng-app="myApp">
  <head lang="en">
    <meta charset="utf-8">
    <title>Getting Started With mcGrid Example</title>
    <link rel="stylesheet" type="text/css" href="bootstrap.min.css" />
    <link rel="stylesheet" type="text/css" href="bootstrap-datetimepicker.min.css" />
    <script type="text/javascript" src="jquery-1.10.2.js"></script>
    <script type="text/javascript" src="bootstrap-datetimepicker.min.js"></script>
    <script type="text/javascript" src="bootstrap.min.js"></script>
    <script type="text/javascript" src="angular.js"></script>
    <script type="text/javascript" src="ui-bootstrap-tpls-0.11.0.js"></script>
    <script type="text/javascript" src="Mc.Grid-1.2.js"></script>
    <script type="text/javascript" src="main.js"></script>
  </head>
  <body ng-controller="MyCtrl">
    <div mc-grid options="gridOptions" source="dataSource">
    </div>
  </body>
</html>
				
// main.js
var app = angular.module('myApp', ['ui.mc.grid','ui.bootstrap']);
app.controller('MyCtrl', function($scope) {
    $scope.dataSource = [{name: "Moroni", age: 50},
                         {name: "Tiancum", age: 43},
                         {name: "Jacob", age: 27},
                         {name: "Nephi", age: 29},
                         {name: "Enos", age: 34}];
    $scope.gridOptions = {columnDefs :[
                            {key: "name", name: "姓名", type: "text" },
                            {key: "age", name: "年龄", type: "text" }
                         ]};
});
