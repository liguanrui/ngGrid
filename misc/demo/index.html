mcGrid

基于angular的一个表格组件，带增删改、分页、搜索组合件

Getting Started

快速指引 (可以结合右边的例子):

  1. 载入 jQuery 和 AngularJS.
  2. 依赖 UI Bootstrap 和 bootstrap-datetimepicker（后期可以考虑加入select2）.
  3. 你声明你的应用程序模块,添加ui.mc.grid、ui.bootstrap:
    angular.module('myApp',['ui.mc.grid','ui.bootstrap']);
    如果你使用 angular seed, 那就加入到 app.js 文件中.
  4. 在 controller 内的 html 里面嵌入:
    <!-- 表格内容 -->
    <div mc-grid options="gridOptions" source="dataSource"></div>
    options：用于配置表格的字段、按钮组、分页.
    source：渲染表格的数据（包含用于分页显示的total）.
  5. 在你 javascript 文件中，嵌入：
    6. $scope.dataSource = {rows: [
                     {name: "Moroni", age: 50},
                     {name: "Tiancum", age: 43},
                     {name: "Jacob", age: 27},
                     {name: "Nephi", age: 29},
                     {name: "Enos", age: 34}]};
					 
$scope.gridOptions = {columnDefs: [
                          {key: "name", name: "姓名", type: "text"},
                          {key: "age", name: "年龄", type: "text"}
                     ]};
  6. 这就是表格的简单使用，如右边的例子:

    <script type="text/javascript" src="bootstrap-datetimepicker.min.js"></script>
    <script type="text/javascript" src="bootstrap.min.js"></script>
    <script type="text/javascript" src="angular.js"></script>
    <script type="text/javascript" src="ui-bootstrap-tpls-0.11.0.js"></script>
    <script type="text/javascript" src="Mc.Grid-1.2.js"></script>
    <script type="text/javascript" src="main.js"></script>
  </head>
  <body ng-controller="GridCtrl">
    <!-- 表头 高级搜索组合框 -->
      <div mc-input options="inputOptions" source="dataSource" param="SeachParam">
        <button data-target="#add-model" data-toggle="modal" class="btn btn-success btn-sm" type="button">新增</button>
        <button data-target="#update-model" data-toggle="modal" class="btn btn-info btn-sm type="button" ng-disabled="chosenItem==null">更新</button>
        <button data-target="#delete-model" data-toggle="modal" class="btn btn-danger btn-sm" type="button" ng-disabled="chosenItem==null">删除</button>
        <button class="btn btn-default btn-sm" type="button" ng-click="doSomeThing()" ng-disabled="chosenItem==null">其他</button>
      </div>

      <!-- 表格 -->
      <div mc-grid options="gridOptions"  source="dataSource"  chosen="chosenItem"   param="SeachParam"></div>
   
  </body>
</html>
				
// main.js
var GridApp=angular.module('GridModule', ['ui.mc.grid','ui.bootstrap']);

function GridCtrl($http,$scope) {
  
  $scope.baseUrl="http://localhost/ngGrid/index.php/book/";
   

  //@TODO这个最好可以脱离smarty,用MVVC的service？
  $scope.op= {"category":{"1":"技术类","2":"其他类","3":"专业类","4":"职业习惯类","5":"管理类","6":"美术类","7":"工具类","15":"小说类"}};
 
  //配置表格
  $scope.gridOptions = {
    combine: [
      {key:"BookId",displayName:"编号"},
      {key:"BookName",displayName:"书名"},
      {key:"BookSort",displayName:"分类",type:"select",arr:$scope.op.category},
    ],
    postUrl : $scope.baseUrl+"getList",
    columnDefs: [           
        {key:"index",displayName:"序号",type:"seqnum"},
        {key:"B_ID",displayName:"编号",type:"id"},
        {key:"B_Sort",displayName:"分类",type:"select",arr:$scope.op.category},
        {key:"B_Name",displayName:"书名"},
        {key:"B_Price",displayName:"价格",cellTemplate:"<span ng-if='row.B_Price'>￥{{row.B_Price}}</span>"},
        //{key:"B_AvailNum",displayName:"库存"},
        {key:"B_BuyTime",displayName:"购买日期",type:"datetime"},
    ],
      addBtn : {postUrl:$scope.baseUrl+"insert"},
      updateBtn : {postUrl:$scope.baseUrl+"update"},
      deleteBtn : {postUrl:$scope.baseUrl+"del"},
      pagination : {perPage:15},
  }


  $scope.doSomeThing = function(){
    alert("打开开发者工具，打印出结果选中结果")
    console.log($scope.chosenItem);
  }
  
}

API

directive attrs describe note
mcInput options 配置组合搜索框（json格式），配置后端请求数据的URL
  • combine: 配置搜索组合框
        属性值（key, name, type, value, arr）
  • postUrl: 获取数据URL
    		•
    source 用于接收返回获取到数据（包含数据列表和数据总数）
  • rows: 数据列表
  • total: 用于分页的总算
  • status: 状态
    		•
    param 用于接收搜索组合框的条件限制参数
  • 按照options.combine的key顺序返回搜索参数
    		•
    mcGrid options 配置组合表格展示（json格式），包含columnDefs
  • columnDefs：表格列配置
  • addBtn：增操作，根据columnDefs生成modal
        属性值（label, nodeId, postUrl）
  • updateBtn：改操作，同addBtn
  • deleteBtn：删操作，同addBtn
  • pagination: 分页
        属性值（postUrl, limit）
    • source 用于接收搜索组合框返回获取到数据（包含数据列表和数据总数）
  • rows: 数据列表
  • total: 用于分页的总算
  • status: 状态
    		•
    param 用于接收搜索组合框搜索组合框的条件限制参数，用于分页
  • 按照options.combine的key顺序返回搜索参数
    		•
    chosen 用于接收返回已选择表格中的一行的数据
  • 按照options.columnDefs的key顺序返回选中一行数据
    		•
    mcModal modalid modal的id /
    modalsize modal的大小 /
    modalhead modal的头部，另外提醒modal的身体和脚部都可以通过transclude加进来 /
    mcEdit column 一般可以通过表格的columnDefs获得 /
    param 接收表单内容，用作后面的内容更新和插入操作 /
    mcdatepicker format 具体可以查看bootstrap-datetimepicker的api，format格式要求一致，例如yyyy/mm/dd HH:ii:ss /
